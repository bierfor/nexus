// actions.ts — Server Actions 2.0
// Type-safety automática, sin boilerplate

"use server";

import { db } from '$lib/database';
import { revalidate } from 'sveltro/cache';

export async function updateProfile(
  formData: FormData
): Promise<{ success: boolean }> {

  const name = formData.get('name') as string;

  await db.user.update({
    where: { id: session.userId },
    data: { name }
  });

  await revalidate('/profile');
  return { success: true };
}

05 — Roadmap

Del concepto
al runtime.

Las fases de construcción de un framework de esta envergadura.

Fase 1 — Compilador
Motor de compilación basado en Svelte 5
Adaptar el compilador de Svelte para emitir HTML puro como bloque base. Integrar Runes como el primitivo de reactividad universal del framework.
Fase 2 — Runtime
Islands Runtime + Hidratación selectiva
Implementar el sistema de directivas client:* y el runtime mínimo de hidratación, cargando JS solo en los componentes marcados.
Fase 3 — Full-stack
Server Actions 2.0 y File-based Router
Construir el servidor basado en Hono/Bun, el sistema de rutas basado en sistema de archivos y el protocolo de Server Actions con inferencia de tipos automática via TypeScript compiler API.
Fase 4 — DX
Tooling, CLI y ecosistema
Dev server con HMR ultrarápido (Vite 6), CLI para scaffolding, adaptadores para Cloudflare, Vercel y Fly.io. Plugin de VS Code / Cursor con soporte completo de Runes.
